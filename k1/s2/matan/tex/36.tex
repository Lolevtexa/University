\begin{eqnarray*}
  \int \dfrac{dx}{\sqrt{x}(x - 1)}
  =\begin{bmatrix}
    t = \sqrt{x}\\
    x = t^2\\
    dx = 2t\, dt
  \end{bmatrix}
  = \int \dfrac{2t}{t(t^2 - 1)}\, dt
  = 2 \int \dfrac{dt}{(t - 1)(t + 1)}
  = \int \dfrac{dt}{t - 1}
  - \int \dfrac{dt}{t + 1}
  = \begin{bmatrix}
    \begin{array}{ll}
      y = t - 1 & z = t + 1\\
      dy = dt& dz = dt
    \end{array}
  \end{bmatrix}
  =\\= \int \dfrac{dy}{y}
  - \int \dfrac{dz}{z}
  = \ln|y|
  - \ln|z|
  + C
  = \ln|\dfrac{y}{z}|
  + C
  = \ln|\dfrac{t + 1}{t - 1}|
  + C
  = \ln|\dfrac{\sqrt{x} - 1}{\sqrt{x} + 1}|
  + C
\end{eqnarray*}

\(
\textbf{Ответ:}
\ln|\dfrac{\sqrt{x} - 1}{\sqrt{x} + 1}|
+ C
\)
